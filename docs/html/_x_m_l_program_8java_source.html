158 public void findModule(String moduleName, Document doc) {
159
160 // Variable para saber si lo hemos encontrado

162 // Obtiene una lista de todos los elementos "module"
163 NodeList modules = doc.getElementsByTagName("module");
164
165 // Itera a través de la lista de módulos.
166 for (int i = 0; i < modules.getLength(); i++) {

168 // Obtenemos el módulo actual
169 Element module = (Element) modules.item(i);

171 // Obtiene el nombre del modulo actual
172 String name = module.getElementsByTagName("name").item(0).getTextContent();
173
174 // Comprobamos si el módulo es igual al que buscamos.

176 // Muestra su información.
